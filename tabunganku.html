Integrasi Google Sheets

Hubungkan database ini ke akun Google Spreadsheet Anda.

Sistem ini dirancang untuk dapat terhubung dengan **Google Apps Script**. Ikuti langkah berikut:

  1. Buat **Google Spreadsheet** baru di Drive Anda.
  2. Buka menu **Ekstensi > Apps Script**.
  3. Tempel (Paste) skrip backend Apps Script yang Anda buat.
  4. Klik **Terapkan > Web App (Aplikasi Web)**, setel akses ke *"Siapa saja (Anyone)"*.
  5. Salin URL Web App dari Google Script dan simpan di kolom berikut untuk diaktifkan:

Kode Backend Google Apps Script

/* KODE GOOGLE APPS SCRIPT BACKEND */
function doPost(e) {
  var params = JSON.parse(e.postData.contents);
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  
  if (params.action == "addSantri") {
    var sheet = ss.getSheetByName("Santri") || ss.insertSheet("Santri");
    if (sheet.getLastRow() == 0) {
      sheet.appendRow(["NIS", "No_Tabungan", "Nama", "Kamar", "Alamat", "Foto", "Saldo"]);
    }
    // Update: Gunakan params.saldo alih-alih hardcode nilai 0
    sheet.appendRow([params.nis, params.notab, params.nama, params.kamar, params.alamat, params.foto, Number(params.saldo || 0)]);
    return ContentService.createTextOutput(JSON.stringify({status: "success", message: "Santri terdaftar"})).setMimeType(ContentService.MimeType.JSON);
  }
  
  if (params.action == "addTransaction") {
    var sheet = ss.getSheetByName("Transaksi") || ss.insertSheet("Transaksi");
    if (sheet.getLastRow() == 0) {
      sheet.appendRow(["ID_Transaksi", "Tanggal", "NIS", "Jenis", "Jumlah", "Keterangan"]);
    }
    sheet.appendRow([params.id, params.tanggal, params.nis, params.jenis, params.jumlah, params.keterangan]);
    
    // Update saldo di sheet Santri
    var santriSheet = ss.getSheetByName("Santri");
    var data = santriSheet.getDataRange().getValues();
    for (var i = 1; i < data.length; i++) {
      if (data[i][0] == params.nis) {
        var currentSaldo = Number(data[i][6]) || 0;
        var change = Number(params.jumlah);
        var newSaldo = params.jenis == "Setor" ? currentSaldo + change : currentSaldo - change;
        santriSheet.getRange(i+1, 7).setValue(newSaldo);
        break;
      }
    }
    return ContentService.createTextOutput(JSON.stringify({status: "success", message: "Transaksi dicatat"})).setMimeType(ContentService.MimeType.JSON);
  }
}
